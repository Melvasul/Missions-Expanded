		required_missions = { italian_nations_italian_conquest	}
		position = 2
		provinces_to_highlight = {
			OR = {
				area = piedmont_area
			}
			NOT = {

			}
		}
		trigger = {
			piedmont_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			if = {
				limit = {
					tag = SAV
				}
				add_prestige = 10
				add_adm_power = 100
				add_dip_power = 100
				add_mil_power = 100
			}
			else = {
				1875 = {
					#Has 1° lvl fort
					if = {
						limit = {
							has_building = fort_15th
						}
						remove_building = fort_15th
						add_building_construction = {
							building = fort_16th
							speed = 0.5
							cost = 0
						}
					}
					#Has 2° lvl fort
					else_if = {
						limit = {
							has_building = fort_16th
						}
						remove_building = fort_16th
						add_building_construction = {
							building = fort_17th
							speed = 0.5
							cost = 0
						}
					}
					#Has 3° lvl fort
					else_if = {
						limit = {
							has_building = fort_17th
						}
						remove_building = fort_17th
						add_building_construction = {
							building = fort_18th
							speed = 0.5
							cost = 0
						}
					}
					else = {
						add_building_construction = {
							building = fort_15th
							speed = 0.5
							cost = 0
						}
					}
				}
				add_country_modifier = {
					name = italian_nations_savoy
					duration = 9125
				}
				piedmont_area = {
					add_province_modifier = {
						name = italian_nations_piedmont
						duration = -1
					}
				}
			}
		}
	}
