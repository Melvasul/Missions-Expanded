	italian_minors_subjucate_piedmont = {
		icon = me_mission_snowy_cannons
		required_missions = {
			italian_minors_italian_conquest
		}
		provinces_to_highlight = {
			OR = {
				area = piedmont_area
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		position = 7
		trigger = {
			piedmont_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			1875 = {
				#Has 1° lvl fort
				if = {
					limit = {
						has_building = fort_15th
					}
					remove_building = fort_15th
					add_building_construction = {
						building = fort_16th
						speed = 0.5
						cost = 0
					}
				}
				#Has 2° lvl fort
				else_if = {
					limit = {
						has_building = fort_16th
					}
					remove_building = fort_16th
					add_building_construction = {
						building = fort_17th
						speed = 0.5
						cost = 0
					}
				}
				#Has 3° lvl fort
				else_if = {
					limit = {
						has_building = fort_17th
					}
					remove_building = fort_17th
					add_building_construction = {
						building = fort_18th
						speed = 0.5
						cost = 0
					}
				}
				else = {
					add_building_construction = {
						building = fort_15th
						speed = 0.5
						cost = 0
					}
				}
			}
			add_country_modifier = {
				name = italian_minors_savoy
				duration = 9125
			}
			piedmont_area = {
				add_province_modifier = {
					name = italian_minors_piedmont
					duration = -1
				}
			}
		}
	}
