	ODA_wild_ginger = {
		icon = mission_assemble_an_army
		required_missions = {
			ODA_rise
		}
		position = 2
		provinces_to_highlight = {
			province_id = 4360
		}
		trigger = {
			if = {
				limit = {
					exists = TKG
				}
				OR = {
					AND = {
						alliance_with = TKG
						TKG = {
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
					}
					AND = {
						is_rival = TKG
						NOT = {
							TKG = {
								has_opinion = {
									who = ROOT
									value = -100
								}
							}
						}
					}
				}
			}
			else = {
				any_country = {
					owns = 4360
					OR = {
						AND = {
							alliance_with = ROOT
							has_opinion = {
								who = ROOT
								value = 150
							}
						}
						AND = {
							is_rival = ROOT
							NOT = {
								has_opinion = {
									who = ROOT
									value = -100
								}
							}
						}
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					exists = TKG
				}
				if = {
					limit = {
						alliance_with = TKG
					}
					set_country_flag = ginger_allied
					add_country_modifier = {
						name = wild_ginger_allied
						duration = 7300
					}
				}
				else_if = {
					limit = {
						is_rival = TKG
					}
					add_country_modifier = {
						name = wild_ginger_rival
						duration = 7300
					}
					add_casus_belli = {
						type = cb_humiliate
						months = 120
						target = TKG
					}
				}
			}
			else = {
				every_country = {
					limit = {
						owns_or_non_sovereign_subject_of = 4360
					}
					if = {
						limit = {
							alliance_with = THIS
						}
						set_country_flag = ginger_allied
						add_country_modifier = {
							name = wild_ginger_allied
							duration = 7300
						}
					}
					else_if = {
						limit = {
							is_rival = THIS
						}
						add_country_modifier = {
							name = wild_ginger_rival
							duration = 7300
						}
						add_casus_belli = {
							type = cb_humiliate
							months = 120
							target = THIS
						}
					}
				}
			}
		}
	}

	ODA_unite_chubu = {
		icon = mission_assemble_an_army
		required_missions = { 
			ODA_wild_ginger 
			ODA_suruga
		}

			}
			NOT = {

			}
		}
		trigger = {
			if = {
				limit = {
					exists = TKG

				}
				chubu_area = {
					all_province = {
						OR = {
							owned_by = ROOT
							owned_by = TKG
						}
					}
				}
				eastern_chubu_area = {
					all_province = {
						OR = {
							owned_by = ROOT
							owned_by = TKG
						}
					}
				}
			}
			else_if = {
				limit = {
					NOT = {
						exists = TKG
					}

				}
				chubu_area = {
					all_province = {
						OR = {
							owned_by = ROOT

						}
					}
				}
				eastern_chubu_area = {
					all_province = {
						OR = {
							owned_by = ROOT
							owned_by = THIS
						}
					}
				}
			}
			else = {
				chubu_area = {
					type = all
					country_or_non_sovereign_subject_holds = ROOT
				}
				eastern_chubu_area = {
					type = all
					country_or_non_sovereign_subject_holds = ROOT
				}
			}
		}
		effect = {
			add_prestige = 20
			kanto_area = {
				limit = {
					NOT = { 
						owned_by = ROOT 
					}
